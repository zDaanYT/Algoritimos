// FUNÇÃO mdc em RECURSIVIDADE

ALGORÍTIMO mdc

FUNÇÃO mdc (INTEIRO: n1, n2)
INÍCIO
	SE n2 == 0
	ENTÃO
		RETORNE n1;
	SENÃO
		RETORNE mdc(n2, n1 % n2);
	
FIM.

// ALGORÍTIMO principal

INÍCIO
	VAR INTEIRO: num1, num2, resultado; // VARIÁVEIS LOCAIS
	
	ESCREVA ("Informe o 1º inteiro: ");
	LEIA num1;

	ESCREVA ("Informe o 2º inteiro: ");
	LEIA num2;
	
	resultado = mdc(num1, num2);
    	ESCREVA("O MDC entre ", num1, " e ", num2, " é: ", resultado);
FIM.

#include <stdio.h>
#include <locale.h>
#include <conio.h>
#include <math.h>

int mdc(int n1, int n2){
        if (n2 == 0){
            return n1;
        }else
            return mdc(n2, n1 % n2);
}

main(){
	setlocale(LC_ALL, "Portuguese");
	int num1, num2;

	printf("Informe o 1º inteiro: ");
	scanf("%d", &num1);

	printf("Informe o 2º inteiro: ");
	scanf("%d", &num2);

	printf("O MDC entre %d e %d é: %d", num1, num2, mdc(num1, num2));
}
