ALGORÍTIMO 

INÍCIO
	VAR REAL: M[n][n];
	INTEIRO: i, j, n;

	ESCREVA ("Informe o valor de n: "); LEIA n;	

	PARA i DE 1 ATÉ n FAÇA
		PARA j DE 1 ATÉ n FAÇA
			ESCREVA ("Informe M[",i,"][",j,"]: "); LEIA M[i][j];
		FIM_PARA;
	FIM_PARA;

	PARA i DE 1 ATÉ n FAÇA
		ESCREVA ("\n");
		PARA j DE 1 ATÉ n FAÇA
			SE i != j
			ENTÃO
				ESCREVA (" ", M[i][j]);
			SENÃO
				ESCREVA (" ");

		FIM_PARA;
	FIM_PARA;	

FIM.

#include <stdio.h>
#include <locale.h>
#define TAM 3 //MATRIZ OBRIGATÓRIAMENTE QUADRADA EX.: 2x2, 3x3...

main(){
    setlocale(LC_ALL, "Portuguese");
    float M[TAM][TAM];
    int i, j, n;

    for(i = 0; i < TAM; i++){
        for(j = 0; j < TAM; j++){
            printf("Informe M[%d][%d]: ", i+1, j+1);
            scanf("%f", &M[i][j]);
        }
    }

    for(i = 0; i < TAM; i++){
        printf("\n");
        for(j = 0; j < TAM; j++){
            if(i != j){
                printf(" %.2f ", M[i][j]);
            }else
                printf("      ");
        }
    }
}
