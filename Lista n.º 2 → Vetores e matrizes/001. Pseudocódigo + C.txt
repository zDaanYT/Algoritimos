#include <stdio.h>
#include <locale.h>
#define TAM 2 // TAMANHO DAS MATRIZES

main(){
    setlocale(LC_ALL, "portuguese");

    int i;
    float A[TAM], B[TAM], C[TAM];

    for(i = 0; i < TAM; i++) {
        printf("\n--------------------\n\n");
        printf("        A[%d]", i + 1);
        printf("\n\n--------------------\n");

        printf("\nInforme o número: ");
        scanf("%f", &A[i]);
    }

    for(i = 0; i < TAM; i++) {
        printf("\n--------------------\n\n");
        printf("        B[%d]", i + 1);
        printf("\n\n--------------------\n");

        printf("\nInforme o número: ");
        scanf("%f", &B[i]);
    }

    printf("\n-----------------------------------------------");

    for(i = 0; i < TAM; i++){
        C[i] = (A[i] + B[i]) / TAM;
        printf("\n O %dº elemento do conjunto C é: %1.1f\n", i + 1, C[i]);
    }

    printf("-----------------------------------------------\n");
}
