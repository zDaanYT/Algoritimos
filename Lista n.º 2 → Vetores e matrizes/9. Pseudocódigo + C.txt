#include <stdio.h>
#include <locale.h>
#define TAM 2 //MATRIZ OBRIGATÓRIAMENTE QUADRADA EX.: 2x2, 3x3...

main(){
	int i, j;
	float A[TAM][TAM], B[TAM][TAM];

	for(i = 0; i < TAM; i++){
			for(j = 0; j < TAM; j++){

			printf("Informe A[%d][%d] = ", i + 1, j + 1);
			scanf("%f", &A[i][j]);

			}
	}


	for(i = 0; i < TAM; i++){
			for(j = 0; j < TAM; j++){

				B[i][j] = A[i][j] / A[i][i];

			}
	}


	for(i = 0; i < TAM; i++){
            printf("\n");
			for(j = 0; j < TAM; j++){

				printf(" %.2f ", B[i][j]);

			}
	}


}
