#include <stdio.h>
#include <locale.h>
#define TAM 2 // TAMANHO DAS MATRIZES QUE SERÃO SOMADAS
#define MAT 4 // TAMANHO DA MATRIZ RESULTADO

main(){
    setlocale(LC_ALL, "portuguese");

    int A[TAM], B[TAM], C[MAT], i;

    for(i = 0; i < TAM; i++) {
        printf("\n--------------------\n\n");
        printf("        A[%d]", i + 1);
        printf("\n\n--------------------\n");

        printf("\nInforme o número: ");
        scanf("%d", &A[i]);
    }

    for(i = 0; i < TAM; i++) {
        printf("\n--------------------\n\n");
        printf("        B[%d]", i + 1);
        printf("\n\n--------------------\n");

        printf("\nInforme o número: ");
        scanf("%d", &B[i]);
    }

    for(i = 0; i < TAM; i++){
            C[i] = A[i];
    }

    for(i = 0; i < TAM; i++){
            C[i + TAM] = B[i];
    }

    printf("\n-----------------------------------------------");

    for(i = 0; i < MAT; i++){
        printf("\n O %dº elemento do conjunto C é: %d \n", i + 1, C[i]);
    }

    printf("-----------------------------------------------\n");
}
