	PARA i DE 1 ATÉ 20

	PARA i DE 1 ATÉ 20

	PARA i DE 1 ATÉ 20

				ESCREVA " ", B[i][j];

#include <stdio.h>

#define L 2
#define C 2

main(){
	int i, j;
	float A[L][C], B[L][C];

	for(i = 0; i < L; i++){
			for(j = 0; j < L; j++){

			printf("Informe A[%d][%d] = ", i + 1, j + 1);
			scanf("%f", &A[i][j]);

			}

	}


	for(i = 0; i < L; i++){
			for(j = 0; j < L; j++){

				B[i][j] = A[i][j] / A[i][i];

			}

	}


	for(i = 0; i < L; i++){
            printf("\n");
			for(j = 0; j < L; j++){

			}

	}


}
