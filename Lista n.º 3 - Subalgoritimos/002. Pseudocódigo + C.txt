ALGORÍTIMO mdc

FUNÇÃO INTEIRO mdc (INTEIRO: n1, n2)
INÍCIO
	VAR INTEIRO: aux; // VARIÁVEIS LOCAIS

	ENQUANTO n2 != 0 FAÇA
		aux = n2;
		n2 = n1 % n2;
		n1 = aux;
	FIM_ENQUANTO;

	RETORNE n1;
FIM.

#include <stdio.h>
#include <locale.h>
#include <conio.h>
#include <math.h>

int mdc(int a, int b){
	int aux;

	while (b != 0){
		aux = b;
		b = a % b;
		a = aux;
	}
	return a;
}

main(){
	setlocale(LC_ALL, "Portuguese");
	int num1, num2, resultado;

	printf("Informe o 1º inteiro: ");
	scanf("%d", &num1);

	printf("Informe o 2º inteiro: ");
	scanf("%d", &num2);

	resultado = mdc(num1, num2);
	printf("O MDC entre %d e %d é: %d", num1, num2, resultado);
}
