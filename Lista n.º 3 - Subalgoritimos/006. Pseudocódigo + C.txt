ALGORÍTMO soma_vetores

FUNÇÃO INTEIRO: soma_vetores(VAR INTEIRO: M[]; INTEIRO: n)
INÍCIO
	VAR INTEIRO: i, soma;

    	i = 1;
    	soma = 0;

	ENQUANTO i <= n FAÇA
		soma = soma + M[i];
		i = i + 1;
	FIM_ENQUANTO;

	RETORNE soma;
FIM.

// ALGORÍTMO principal

INÍCIO
    	VAR INTEIRO: i, n, M[100];

	ESCREVA("Informe o valor de n: "); LEIA n;

	PARA i DE 1 ATÉ n FAÇA
        	ESCREVA("Digite o valor M[",i,"]: "); LEIA M[i];
	FIM_PARA;

    	ESCREVA("A soma do Vetor é: ", soma_vetores(M, n));
FIM.

#include <stdio.h>
#include <locale.h>
#include <conio.h>
#include <math.h>

int soma_vetores(int *M, int n){
    int i = 0, soma = 0;

    do{
        soma = soma + M[i];
		i++;
    } while(i < n);

    return soma;
}

main(){
    int i, n, M[100];
    setlocale(LC_ALL, "portuguese");

    printf("Informe o valor de n: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++){
        printf("Digite o valor M[%d]: ", i + 1);
        scanf("%d", &M[i]);
    }

    printf("A soma do Vetor é: %d", soma_vetores(M, n));
}
