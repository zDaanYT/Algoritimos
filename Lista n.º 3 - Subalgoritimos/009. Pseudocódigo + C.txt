ALGORÍTMO vetor_maior

FUNÇÃO REAL: verifique(VAR REAL: M[n]; INTEIRO: n)
INÍCIO
    VAR INTEIRO: i;
    VAR REAL: maior;

    maior = M[1];

    PARA i DE 2 ATÉ n FAÇA
        SE M[i] > maior 
        ENTÃO
            maior = M[i];
    FIM_PARA;

    RETORNE maior;
FIM.

#define TAM 100

float verifique(float *M, int n){
    int i;
    float maior;

    maior = M[0];

    for(i = 1; i < n; i++){ 
        if(M[i] > maior){
            maior = M[i];  
        }
    }
    return maior;
}
