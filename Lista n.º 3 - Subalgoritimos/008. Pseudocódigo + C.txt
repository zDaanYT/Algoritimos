	VAR INTEIRO: bin[8], i;

	PARA i DE 1 ATÉ 8 FAÇA
		bin[i] = 0;
	FIM_PARA;

	i = 8;
	
	ENQUANTO n > 0 FAÇA
		bin[i] = n % 2;
		n = INT(n / 2);
		i = i - 1;
	FIM_ENQUANTO;
	
	PARA i DE 1 ATÉ 8 FAÇA
		ESCREVA (" ", bin[i]);
	FIM_PARA;
FIM.

#include <stdio.h>
#include <locale.h>
#include <conio.h>
#include <math.h>

void binario(int n){
	int bin[8], i;

	for(i = 0; i <= 7; i++){
		bin[i] = 0;
	}

	i = 7;

	while(n > 0){
		bin[i] = n % 2;
		n = n / 2;
		i--;
	}

	for(i = 0; i <= 7; i++){
		printf("%d", bin[i]);
	}
}

main(){
	int n;
	setlocale(LC_ALL, "Portuguese");

    printf("Informe o valor de n (0 a 255): ");
    scanf("%d", &n);

    if (n < 0 || n > 255) {
        printf("Valor inválido! Insira um número entre 0 e 255.\n");
    } else {
        binario(n);
    }
}
