#include <stdio.h>
#include <locale.h>
#include <conio.h>
#include <math.h>

int soma_cubo(int termo1, int termoN){
    int i = 0, soma = 0, n;

    if(termo1 == 0 && termoN == 0){
        return soma;
    } else {
        n = termoN - termo1 + 1;

        for(i = 0; i <= n; i++){
            soma =+ i * i * i;
        }
    }
    return soma;
}
